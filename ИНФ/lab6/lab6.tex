\newcommand{\ElemLabelL}[4]{
  \begin{minipage}{2.2cm}
   \makebox[\textwidth][lcm]{%
    \textbf{#3}% 元素符号左对齐
    \hfill
   \makebox[0.5cm][r]{%
    \begin{tabular}{@{}r@{}}
      #1 \\ 
      #2    
    \end{tabular}%
  }%
}
    \linebreak\linebreak
    {{#4}}
  \end{minipage}
}

\newcommand{\ElemLabelR}[4]{
  \begin{minipage}{2.2cm}
    \makebox[\textwidth][lcm]{%
      \makebox[0.5cm][l]{%
        \begin{tabular}{@{}l@{}} % 左对齐
          #1 \\ % 第一行显示 #1
          #2    % 第二行显示 #2
        \end{tabular}%
      }%
      \hfill
      \textbf{#3} % 元素符号
    }
    \linebreak \linebreak
    \makebox[\textwidth][r]{{{#4}}} % 元素名称右对齐
  \end{minipage}
}
